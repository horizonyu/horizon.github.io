##与服务器进行交互的代码

  //如果服务器有新版本提示用户进行更新

    new Thread() {
        @Override
        public void run() {
            long startTime = System.currentTimeMillis();
            try {
                //1.向服务器发送请求数据
                URL url = new URL("http://169.254.161.57:8080/update74.json");

                //2.开启一个链接
                HttpURLConnection connection = (HttpURLConnection) url.openConnection();

                //3.设置请求的一些参数
                connection.setConnectTimeout(3000);
                connection.setReadTimeout(2000);
                //  connection.setRequestMethod("GET");//默认是GET 方式，所以此处可以省略

                //4.判断获得的响应码  200表示连接成功
                int responseCode = connection.getResponseCode();
                if (responseCode == 200) {
                    //5.获取数据流
                    InputStream inputStream = connection.getInputStream();

                    //6. 将数据以流的形式转化为字符串的形式
                    String json = StreamUtil.stream2string(inputStream);

                    msg = new Message();

                    //获取json实例，将json格式的数据解析出来
                    try {
                        JSONObject jsonObject = new JSONObject(json);

                        //版本名称
                        mVersionName = jsonObject.getString("versionName");
                        //版本号
                        mVersionCode = jsonObject.getString("versionCode");
                        //版本描述
                        mVersionDes = jsonObject.getString("versionDes");
                        //软件的下载地址
                        mDownloadUrl = jsonObject.getString("downloadUrl");

                        Log.i("SplashActivity", mVersionName);
                        Log.i("SplashActivity", mVersionCode);
                        Log.i("SplashActivity", mVersionDes);
                        Log.i("SplashActivity", mDownloadUrl);

                        //如果服务器中的软件的版本号大于手机中的版本号，则进行更新提醒
                        if (Integer.parseInt(mLocalVersionCode) < Integer.parseInt(mVersionCode)) {
                            //在此处提示用户进行的提醒
                            msg.what = UPDATE_VERSION;

                        } else {
                            //进入程序的主界面
                            msg.what = ENTER_HOME;
                        }

                    } catch (JSONException e) {
                        e.printStackTrace();
                    }

                }

            } catch (MalformedURLException e) {
                e.printStackTrace();
            } catch (IOException e) {
                e.printStackTrace();
            } finally {
                //为了保证引导页的效果，所以需要满足在引导页停留4S中的要求
                long endTime = System.currentTimeMillis();
                if ((endTime - startTime) < 4000) {
                    try {
                        Thread.sleep(4000 - (endTime - startTime));
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
                //利用Handler的实例，调用sendMessage()方法，将msg信息返回Handler中
                mHandler.sendMessage(msg);
            }
        }
    }.start();




public static String stream2string(InputStream inputStream) throws IOException {
    //1. 将流中的数据读取之后放入缓存中，然后 一次性转换为字符串
    ByteArrayOutputStream bos = new ByteArrayOutputStream();

    //2. 创建一个字符数组用于暂存读取流中的数据
    byte[] bytes = new byte[1024];

    //3.定义一个临时变量，记录读取的数据
    int temp = -1;

    //4.定义一个循环，依次读入数据

        while ((temp = inputStream.read(bytes) )!= -1){
            bos.write(bytes,0,temp);

        }
        //5.返回读取的数据
          return  bos.toString();

    }

###Sharedpreferences 静态变量不属于任何对象

1. 当需要在一个Activity中处理完数据后，将数据保存下来，可以将其放入到SharedPreferences 中，下一次在此启动此Activity时，将数据读取出来。

     private static SharedPreferences sp;

/**
 *
 * @param context  获取上下文
 * @param key      节点
 * @param defValue 节点对应的值
 * @return         返回点击的状态
 */
//获取状态值
public static boolean getBoolean(Context context,String key,boolean defValue)
{
    if(sp == null)
    sp = context.getSharedPreferences("config", Context.MODE_PRIVATE);

    boolean aBoolean = sp.getBoolean(key, defValue);

    return aBoolean;
}

/**
 *
 * @param context  传入上下文
 * @param key      获取相应的键
 * @param value    获取相应的键对应的值
 *
 */
   public static void putBoolean(Context context,String key,boolean value)
    {
        if(sp == null)
            sp = context.getSharedPreferences("config", Context.MODE_PRIVATE);

        sp.edit().putBoolean(key,value).commit();
   }
